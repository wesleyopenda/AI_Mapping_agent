class Student:
    def __init__(self, studentName, studentID):
        self.studentName = studentName
        self.studentID = studentID

    def setAge(self, age):
        self.age = age

    def setMarks(self, marks):
        self.marks = marks

    def __str__(self):
        if(hasattr(self, "age") and hasattr(self, "marks")):
            return(self.studentName + ": " + str(self.studentID) + " is "
                   + str(self.age) + " years old and has received "
                   + str(self.marks) + "/100 this year.")
        else:
            return ("Please set the age and marks for " + self.studentName)

s1 = Student("Wes", 7)
s1.setAge(24)
s1.setMarks(95)
print(s1)

s2 = Student("Glo", 1)
print(s2)
